4. Moyenne, médiane, quartiles et écart interquartile

import math

def mediane(liste):
    data = sorted(liste)
    N = len(data)
    if N % 2 == 1:
        return data[N // 2]
    else:
        return (data[N // 2 - 1] + data[N // 2]) / 2

def quartiles(liste):
    data = sorted(liste)
    N = len(data)
    rang_Q1 = math.ceil(N / 4)
    rang_Q3 = math.ceil(3 * N / 4)
    Q1 = data[rang_Q1 - 1]
    Q3 = data[rang_Q3 - 1]
    return Q1, Q3, Q3 - Q1

for numero, serie in [(1, serie1), (2, serie2), (3, serie3)]:
    moyenne = sum(serie) / len(serie)
    med = mediane(serie)
    Q1, Q3, IQR = quartiles(serie)
    print("Série", numero)
    print("  moyenne =", round(moyenne, 2))
    print("  médiane =", med)
    print("  Q1 =", Q1, "Q3 =", Q3, "écart interquartile =", IQR)

5. Écart-type

def ecart_type(liste):
    m = sum(liste) / len(liste)
    variance = 0
    for x in liste:
        variance += (x - m) ** 2
    variance = variance / len(liste)
    return math.sqrt(variance)

print("Écart-type série 1 :", round(ecart_type(serie1), 2))
print("Écart-type série 2 :", round(ecart_type(serie2), 2))
print("Écart-type série 3 :", round(ecart_type(serie3), 2))

9. Application de K-means avec scikit-learn

Ici, on choisit finalement k = 2.

from sklearn.cluster import KMeans

kmeans = KMeans(n_clusters=2, n_init=20, random_state=0)
kmeans.fit(eleves)

labels = kmeans.labels_
centres = kmeans.cluster_centers_

print("Étiquette de chaque élève :", labels)
print("Centres des 2 groupes :")
for i, centre in enumerate(centres):
    print("Groupe", i, ":", [round(x, 2) for x in centre])
